func authRequired() gin.HandlerFunc {
    return func(c *gin.Context) {
        session := sessions.Default(c)
        user := session.Get("user")

        if user == nil {
            c.JSON(401, gin.H{"message": "unauthorized"})
            c.Abort()
            return
        }

        c.Next()
    }
}
